B 問題

隣接 2 項を見るためには zipWith が便利です。

solve :: StateT BS.ByteString IO ()
solve = do
  _ <- line'
  !xs <- U.map (,True) <$> intsU'
  !ys <- U.map (,False) <$> intsU'
  let !zs = U.map snd $ U.modify (VAI.sortBy (comparing fst)) $ xs U.++ ys
  let !res = U.or $ U.zipWith (&&) zs (U.tail zs)
  printYn res
